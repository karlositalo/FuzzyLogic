#Matriz Condicional
def MaCond(A):
    n = A.shape[0]
    C = np.zeros((n,n))
    for i in range(n):
        for j in range(n):
            C[i,j] = A[j,i] / max(A[j,i],A[i,j])
    return C

#Vetor de ordens
def V_Ordem(M):
    n = M.shape[0]
    v = np.zeros((n))
    for i in range(n):
        v[i] = min(M[i,:])
    return v

    v = np.zeros((c,1))

def ProMarg(Mat,P): 
    l,c = Mat.shape
    Pm = np.zeros((l))
    for j in range(l):
        for i in range(l):
            Pm[j] = Pm[j] + Mat[i,j]*P[i]
    return Pm

def PoPrF(F,U,P,PM):
    l,c = F.shape
    MP = np.zeros((c,l))
    for i in range(l):
        for k in range(c):
            soma = 0
            for j in range(c):
                soma = soma + (F[i,j]*U[j,k]*P[j])
            MP[k,i] = (soma/PM[k])
